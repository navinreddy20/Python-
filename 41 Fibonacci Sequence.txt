def fib(n):

    a = 0
    b = 1
    
    if n < 0:
      print("incorrect input")
      
    elif n == 0:
      return 0
      
    elif n == 1:
        print(a)

    else:
        print(a)
        print(b)

    for i in range(2,n):
        c = a + b
        a = b
        b = c
        print(c)

fib(9)
